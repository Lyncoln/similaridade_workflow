csvData Each line is a different piece of the data. Line 1: Country Name Line 2: Income share of lowest 20% data Line 3: Year range for income share Line 4: Literacy rate Line 5: Literacy rate years Please note that if data begins before 1985 or ends after 2005, the chds value in googleChartsURLBuilder will need to be changed from: sb.append("&chds=1985,2005,0,100,1985,2005,0,100"); to sb.append("&chds=STARTYEAR,ENDYEAR,0,100,STARTYEAR,ENDYEAR,0,100");

Script

StringBuilder sb = new StringBuilder();

sb.append("http://chart.apis.google.com/chart?");

/**  Axis ranges**/
sb.append("chxr=0,1985,2005");
/** Axis label styles. **/
sb.append("&chxs=0,6B687A,13,0.5,l,676767|1,676767,13,0,l,676767");
/** Visible axes **/
sb.append("&chxt=x,y,x,y&chxl=2:||Year||3:||Percentage||");
/** Chart size **/
sb.append("&chs=660x450");
/** Chart type **/
sb.append("&cht=lxy");
/** Chart colors **/
sb.append("&chco=AA0033,0000FF");
/** Scale for text format with custom range. **/
sb.append("&chds=1985,2005,0,100,1985,2005,0,100");
/** Chart data string. NB: complicated. **/
sb.append("&chd=t:");
sb.append(literacyDataYears);
sb.append("|");
sb.append(literacyData);
sb.append("|");
sb.append(lowest20DataYears);
sb.append("|");
sb.append(lowest20Data);
/** Chart Legend **/
sb.append("&chdl=Literacy+Rates|Income+Share");
/** Grid lines **/
sb.append("&chg=5,5,2,2");
/** Line styles **/
sb.append("&chls=1,2,2|1,3,4");
/** Line markers **/
sb.append("&chm=B,EFEFEFC6,0,0,0");
/** Chart title **/
sb.append("&chtt=Literacy+Rates+and+Income+Share+of+Lowest+20%+(");
sb.append(countryName);
sb.append(")");

graphURL = sb.toString();
Get_Image_From_URL localworker Grabs the image from the URL created by googleChartsURLBuilder.

Script

if ((url == void) || (url == null)) {
	throw new RuntimeException("The url must be specified");
}

URL inputURL = null;
if (base != void) {
	inputURL = new URL(new URL(base), url);
} else {
	inputURL = new URL(url);
}

int bytesRead = 0;
int totalBytesRead = 0;
InputStream is = inputURL.openStream();
ByteArrayOutputStream os = new ByteArrayOutputStream();
byte[] buffer = new byte[2048];
while (true) {
	bytesRead = is.read(buffer);
	if (bytesRead == -1) {
		break;	
	}

	os.write(buffer, 0, bytesRead);
}

image = os.toByteArray();
is.close();
os.close();
